    <script>
        function doSearch() {
            const o = document.getElementById('orig').value, d = document.getElementById('dest').value, dd = document.getElementById('dDate').value, rd = document.getElementById('rDate').value;
            if(o && d && dd) window.open(`https://search.aviasales.com/?origin_iata=${o}&destination_iata=${d}&depart_date=${dd}&return_date=${rd}&marker=HkZPtDdO`, '_blank');
            else alert("Please fill at least Origin, Destination and Departure Date.");
        }
