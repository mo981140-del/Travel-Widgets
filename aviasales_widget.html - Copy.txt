    <script>
        document.getElementById('searchBtn').onclick = function() {
            const origin = document.getElementById('origin').value;
            const dest = document.getElementById('destination').value;
            const dep = document.getElementById('depDate').value;
            const ret = document.getElementById('retDate').value;
            if(origin && dest && dep) {
                const url = `https://search.aviasales.com/?origin_iata=${origin}&destination_iata=${dest}&depart_date=${dep}&return_date=${ret}&marker=HkZPtDdO`;
                window.open(url, '_blank');
            } else { alert("Please fill origin, destination and departure date."); }
        };
